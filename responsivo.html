Design Responsivo e Adaptação

O site deve ser acessível e utilizável em todos os dispositivos, desde celulares até desktops, adaptando seu layout dinamicamente.

Exemplo: Teste de Responsividade

Este bloco se ajusta automaticamente ao tamanho da tela.

Código-Fonte Utilizado

HTML:
<div id="caixa-responsiva" class="caixa-responsiva">Este bloco se ajusta automaticamente ao tamanho da tela.</div>
<p id="mensagem-responsiva"></p>

JS:
function atualizarResponsividade() {
  const largura = window.innerWidth;
  const caixa = document.getElementById('caixa-responsiva');
  const mensagem = document.getElementById('mensagem-responsiva');

  if (largura <= 480) {
    caixa.style.backgroundColor = '#FFB6C1'; // Rosa claro
    mensagem.innerText = "Modo Celular Ativo";
  } else if (largura <= 768) {
    caixa.style.backgroundColor = '#ADD8E6'; // Azul claro
    mensagem.innerText = "Modo Tablet Ativo ";
  } else {
    caixa.style.backgroundColor = '#90EE90'; // Verde claro
    mensagem.innerText = "Modo Desktop Ativo";
  }
}

window.addEventListener('resize', atualizarResponsividade);
window.addEventListener('load', atualizarResponsividade);
