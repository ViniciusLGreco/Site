Pagamentos e Compras Acessíveis

O processo de compra deve ser simples, seguro e acessível, com campos claros e confirmação por voz e mensagem visual.

Exemplo: Formulário de Pagamento

Código-Fonte Utilizado

HTML:
<label for="cartao">Número do Cartão:</label>
<input type="text" id="cartao" placeholder="0000 0000 0000 0000">
<button onclick="confirmarCompra()">Finalizar Compra</button>
<p id="mensagem-pagamento"></p>

JS:
function confirmarCompra() {
    const cartao = document.getElementById("cartao");
    const mensagem = document.getElementById("mensagem-pagamento");
    const valor = cartao.value.replace(/\D/g, "");
  
    if (valor.length < 16) {
      const textoErro = "Número do cartão inválido. Insira os 16 dígitos.";
      mensagem.innerText = textoErro;
      mensagem.style.color = "#dc3545";
      mensagem.setAttribute("role", "alert");
      cartao.focus();
  
      const vozErro = new SpeechSynthesisUtterance(textoErro);
      window.speechSynthesis.speak(vozErro);
      return;
    }
    if (valor.length > 16) {
      const textoErro = "Número do cartão inválido. Insira apenas 16 dígitos.";
      mensagem.innerText = textoErro;
      mensagem.style.color = "#dc3545";
      mensagem.setAttribute("role", "alert");
      cartao.focus();
  
      const vozErro = new SpeechSynthesisUtterance(textoErro);
      window.speechSynthesis.speak(vozErro);
      return;
    }
  
    const textoSucesso = "Compra realizada com sucesso!";
    mensagem.innerText = textoSucesso;
    mensagem.style.color = "#198754";
    const voz = new SpeechSynthesisUtterance(textoSucesso);
    window.speechSynthesis.speak(voz);
  }
